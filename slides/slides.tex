\section{Problema}

\begin{frame}{Problema}
    \begin{block}{Pergunta}
        Quais fatores estão associados ao parto cesáreo na Bahia e como isso varia entre municípios?
    \end{block}

    \begin{itemize}
        \item Base: SINASC (DATASUS): Declarações de Nascidos Vivos
        \item Recorte: Bahia, 2022 a 2024
        \item Alvo: tipo de parto (vaginal vs.\ cesárea)
        \item \textbf{Foco:} determinantes e variação territorial; \emph{não} predição clínica individual
    \end{itemize}
\end{frame}

\begin{frame}{Objetivo analítico}
                \begin{enumerate}
                    \item Treinar e validar um classificador
                    \item Extrair importância de variáveis
                    \item Comparar padrões entre municípios
                    \item Confrontar com literatura epidemiológica
                \end{enumerate}

    \vspace{0.4em}
\end{frame}

\section{Jornada metodológica}

\begin{frame}{Como chegamos ao modelo}
    \textbf{Abordagem iterativa:} cada etapa informa a próxima.

    \vspace{0.4em}
    \begin{enumerate}
        \item Definir pergunta, recorte geográfico e temporal
        \item Unificar e limpar a base; excluir registros inválidos
        \item Remover variáveis de vazamento e controle administrativo
        \item Selecionar determinantes plausíveis, \textbf{poucas no início}
        \item Treinar Random Forest, medir desempenho, interpretar importâncias
        \item Iterar: ampliar ou podar variáveis; comparar se melhorou ou piorou
        \item Validar achados com literatura e variar por município
    \end{enumerate}
\end{frame}

\begin{frame}{Target encoding territorial}
    \textbf{Problema:} município (418) e CNES (1.022) estouram one-hot --- importância diluída.

    \vspace{0.4em}
    \textbf{Solução:} substituir por taxa histórica de cesárea no treino (smoothing $m=10$):
    \[
        \hat{p}_c = \frac{n_c \cdot \bar{y}_c + m \cdot \bar{y}}{n_c + m}
    \]

    \vspace{0.4em}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Interpretação}
            \begin{itemize}
                \item Efeito \textbf{contextual/ecológico}, não causal individual
                \item CNES: prática da unidade onde pariu
                \item Município: ambiente territorial mais amplo
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Cuidados}
            \begin{itemize}
                \item Fit apenas no treino (sem vazamento)
                \item 224 CNES com taxa 100\% (0{,}4\% dos nascimentos) --- pode inflar AUC
                \item Não confundir com ``hospital decide sozinho''
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\section{Resultados}

\begin{frame}{Evolução do desempenho (hold-out 20\%)}
    \begin{table}
        \centering
        \scriptsize
        \begin{tabular}{llrrrr}
            \toprule
            Exp. & Versão & Feat. & AUC-ROC & F1 & Acurácia \\
            \midrule
            v0 & baseline (4 vars.) & 421 & 0{,}694 & 0{,}624 & 0{,}642 \\
            v2 & + obstétrico / social & 447 & 0{,}741 & 0{,}673 & 0{,}682 \\
            v4 & + gestação / Kotelchuck & 453 & 0{,}748 & 0{,}678 & 0{,}686 \\
            v5 & + assistência & 464 & 0{,}849 & 0{,}762 & 0{,}761 \\
            v6 & + território (one-hot) & 1.435 & 0{,}878 & 0{,}793 & 0{,}792 \\
            ablation & v6 sem município & 1.017 & 0{,}872 & 0{,}790 & 0{,}788 \\
            \textbf{atual} & \textbf{v6 + target encode} & \textbf{53} & \textbf{0{,}883} & \textbf{0{,}800} & \textbf{0{,}797} \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.3em}
    \begin{itemize}
        \item Maior salto clínico-assistencial: v4 $\rightarrow$ v5 (\texttt{TPNASCASSI})
        \item Target encoding: +0{,}006 AUC vs.\ v6 one-hot, com 27$\times$ menos colunas
        \item Métricas altas \textbf{não} são o produto; servem de suporte metodológico
    \end{itemize}
\end{frame}

\begin{frame}{Importâncias --- modelo atual (target encoding)}
    \textbf{Random Forest · v6 · 13 variáveis · 53 \emph{features}}

    \vspace{0.4em}
    \begin{table}
        \centering
        \small
        \begin{tabular}{lr}
            \toprule
            Determinante & Importância relativa \\
            \midrule
            Taxa de cesárea do estabelecimento (CNES) & 23{,}0\% \\
            Idade da mãe & 14{,}5\% \\
            Taxa de cesárea do município de residência & 13{,}4\% \\
            Cesáreas anteriores & 9{,}7\% \\
            Profissional: enfermeiro/obstetra (\texttt{TPNASCASSI}) & 9{,}3\% \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.4em}
    \begin{itemize}
        \item Com one-hot, CNES e município \textbf{não} apareciam no top --- encoding tornou o efeito territorial legível
        \item Idade, histórico obstétrico: coerentes com literatura
        \item CNES $>$ município: unidade de parto pesa mais que o território agregado
    \end{itemize}
\end{frame}

\begin{frame}{O que aprendemos nas iterações}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Ganhos plausíveis}
            \begin{itemize}
                \item Paridade, escolaridade, gestação (v1--v4)
                \item Kotelchuck $>$ contagem bruta de consultas
                \item Target encoding: territorial legível e modelo enxuto
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Pontos de atenção}
            \begin{itemize}
                \item \texttt{TPNASCASSI} infla AUC --- variável de processo?
                \item Taxa do CNES pode capturar unidades extremas (100\% cesárea)
                \item Associação $\neq$ causalidade individual
            \end{itemize}
        \end{column}
    \end{columns}

    \vspace{0.5em}
    \begin{alertblock}{Ablation município (one-hot v6)}
        Remover \texttt{CODMUNRES}: $-$418 features, AUC cai apenas 0{,}006 --- sinal municipal redundante com CNES.
    \end{alertblock}
\end{frame}
