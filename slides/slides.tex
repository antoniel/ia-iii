\section{Problema}

\begin{frame}{Problema}
    \begin{block}{Pergunta}
        Quais fatores estão associados ao parto cesáreo na Bahia e como isso varia entre municípios?
    \end{block}

    \begin{itemize}
        \item Base: SINASC (DATASUS): Declarações de Nascidos Vivos
        \item Recorte: Bahia, 2022 a 2024
        \item Alvo: tipo de parto (vaginal vs.\ cesárea)
        \item \textbf{Foco:} determinantes e variação territorial; \emph{não} predição clínica individual
    \end{itemize}
\end{frame}

\begin{frame}{Objetivos}
    \begin{enumerate}
        \item Desenvolver e validar um modelo de classificação supervisionada
        \item Identificar a importância relativa dos determinantes do parto cesáreo
        \item Descrever a variação territorial entre municípios
        \item Confrontar os achados com a literatura epidemiológica
    \end{enumerate}
\end{frame}

\section{Construção do modelo}

\begin{frame}{Metodologia}
    \textbf{Estrutura da apresentação:}
    \begin{enumerate}
        \item \textbf{Construção do modelo}: seleção de dados, variáveis e decisões analíticas
        \item \textbf{Resultados}: desempenho, importância relativa das variáveis e confronto com a literatura
    \end{enumerate}

    \vspace{0.5em}
    \textbf{Estratégia de modelagem:}
    \begin{itemize}
        \item Partir de um conjunto reduzido de determinantes e expandir por blocos temáticos
        \item Avaliar o ganho discriminativo a cada etapa e justificar inclusão ou exclusão de variáveis
        \item Excluir indicadores de processo assistencial quando o objetivo é interpretar determinantes anteparto
    \end{itemize}
\end{frame}

\begin{frame}{Codificação de alta cardinalidade e Robson}
    \textbf{Variáveis territoriais e clínicas} (município, estabelecimento, Robson) codificadas pela taxa de cesárea suavizada no conjunto de treino:
    \[
        \hat{p}_c = \frac{n_c \cdot \bar{y}_c + m \cdot \bar{y}}{n_c + m}, \quad m = 10
    \]

    \vspace{0.3em}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Classificação de Robson}
            \begin{itemize}
                \item Instrumento recomendado pela OMS e adotado pelo Ministério da Saúde
                \item Estratifica gestantes por paridade, histórico, apresentação e idade gestacional
                \item Padrão em estudos populacionais com SINASC
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Limitações interpretativas}
            \begin{itemize}
                \item Efeito contextual/ecológico, não necessariamente causal ao nível individual
                \item Robson agrega informações clínicas já presentes no modelo
                \item Parâmetros estimados apenas no treino, para evitar vazamento
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Exclusão do profissional assistente ao parto}
    \textbf{Variável considerada:} categoria do profissional que assistiu o parto (médico, enfermeiro/obstetra, etc.).

    \vspace{0.4em}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Fundamentação da exclusão}
            \begin{itemize}
                \item Indicador de \textbf{processo assistencial}, registrado no momento do parto
                \item Associado a ganho discriminativo de $\approx$5 pontos percentuais de AUC
                \item Incompatível com a pergunta sobre determinantes anteparto do desfecho
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Modelo adotado}
            \begin{itemize}
                \item Mantém histórico obstétrico, perfil demográfico e mobilidade territorial
                \item Compromisso metodológico: menor AUC, maior interpretabilidade epidemiológica
                \item Variáveis adicionais recuperam parte modesta do desempenho perdido
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Desempenho do modelo final}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Métricas no conjunto de teste}
            \begin{itemize}
                \item AUC-ROC: 0{,}879
                \item F1: 0{,}791
                \item Acurácia: 0{,}793
            \end{itemize}

        \begin{column}{0.48\textwidth}

                \item Importâncias relativas (MDI) indicam ordem de magnitude, não efeito causal

\begin{frame}{Síntese dos achados}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Convergência com a literatura}
            \begin{itemize}
                \item Histórico obstétrico (cesárea anterior e parto vaginal prévio)
                \item Idade e escolaridade materna
                \item Estratificação de Robson e heterogeneidade entre unidades de saúde
                \item Determinantes sociais com peso inferior ao bloco clínico-territorial
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Contribuições específicas}
            \begin{itemize}
                \item Recorte populacional: Bahia, 2022--2024 ($\approx$50\% de cesáreas)
                \item Estabelecimento de saúde supera município na importância relativa
                \item Exclusão do profissional assistente altera o ranking e a leitura causal
                \item Mobilidade territorial (nascer onde reside): associação modesta
            \end{itemize}
        \end{column}
    \end{columns}
